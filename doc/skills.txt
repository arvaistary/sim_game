1. Объект skillModifiers + функция пересчёта
JavaScript/**
 * skillModifiers.js
 * Глобальные модификаторы, рассчитанные на основе уровней всех навыков
 */

export let skillModifiers = {
    // === Шкалы ===
    hungerDrainMultiplier: 1.0,        // множитель падения голода
    energyDrainMultiplier: 1.0,
    stressGainMultiplier: 1.0,
    moodRecoveryMultiplier: 1.0,
    healthDecayMultiplier: 1.0,        // скорость старения / падения здоровья

    // === Работа и экономика ===
    salaryMultiplier: 1.0,             // общий множитель ЗП
    workEfficiencyMultiplier: 1.0,     // эффективность рабочего периода
    shopPriceMultiplier: 1.0,          // цены в магазине
    investmentReturnMultiplier: 1.0,

    // === Обучение ===
    learningSpeedMultiplier: 1.0,      // скорость прохождения курсов и книг

    // === Дом и комфорт ===
    homeComfortMultiplier: 1.0,        // эффективность мебели
    dailyExpenseMultiplier: 1.0,       // ежедневные расходы

    // === События ===
    positiveEventChanceBonus: 0.0,     // дополнительный шанс положительных событий
    negativeEventPenaltyReduction: 0.0,// снижение штрафа от негативных событий

    // === Отношения ===
    relationshipGainMultiplier: 1.0,

    // === Хобби и пассивный доход ===
    hobbyIncomeMultiplier: 1.0,
    passiveIncomeBonus: 0,             // фиксированная прибавка в рублях в месяц

    // === Специальные ===
    maxEnergyBonus: 0,
    agingSpeedMultiplier: 1.0,
    foodRecoveryMultiplier: 1.0,
};

// =============================================
// Основная функция пересчёта модификаторов
// =============================================

export function recalculateSkillModifiers(skillLevels) {
    // skillLevels — объект вида { timeManagement: 7, professionalism: 4, ... }

    // Сбрасываем все модификаторы к базовым значениям
    Object.keys(skillModifiers).forEach(key => {
        if (typeof skillModifiers[key] === 'number') {
            skillModifiers[key] = (key.includes('Bonus') || key.includes('passiveIncome')) ? 0 : 1.0;
        }
    });

    // Проходим по всем навыкам и применяем их эффекты
    ALL_SKILLS.forEach(skill => {
        const level = skillLevels[skill.key] || 0;
        if (level <= 0) return;

        const lvl = level; // для удобства

        switch (skill.key) {

            // ====================== БАЗОВЫЕ НАВЫКИ ======================
            case 'timeManagement':
                skillModifiers.energyDrainMultiplier *= (1 - lvl * 0.025);
                skillModifiers.stressGainMultiplier *= (1 - lvl * 0.02);
                skillModifiers.workEfficiencyMultiplier *= (1 + lvl * 0.015);
                skillModifiers.learningSpeedMultiplier *= (1 + lvl * 0.03);
                break;

            case 'communication':
                skillModifiers.positiveEventChanceBonus += lvl * 0.012;
                skillModifiers.relationshipGainMultiplier *= (1 + lvl * 0.04);
                break;

            case 'financialLiteracy':
                skillModifiers.shopPriceMultiplier *= (1 - lvl * 0.018);
                skillModifiers.investmentReturnMultiplier *= (1 + lvl * 0.035);
                skillModifiers.passiveIncomeBonus += lvl * 80;
                break;

            case 'healthyLifestyle':
                skillModifiers.hungerDrainMultiplier *= (1 - lvl * 0.022);
                skillModifiers.healthDecayMultiplier *= (1 - lvl * 0.03);
                skillModifiers.foodRecoveryMultiplier *= (1 + lvl * 0.04);
                break;

            case 'adaptability':
                skillModifiers.negativeEventPenaltyReduction += lvl * 0.04;
                break;

            case 'discipline':
                skillModifiers.learningSpeedMultiplier *= (1 + lvl * 0.025);
                break;

            case 'physicalFitness':
                skillModifiers.maxEnergyBonus += lvl * 4;
                skillModifiers.energyDrainMultiplier *= (1 - lvl * 0.01);
                break;

            case 'emotionalIntelligence':
                skillModifiers.moodRecoveryMultiplier *= (1 + lvl * 0.045);
                skillModifiers.negativeEventPenaltyReduction += lvl * 0.025;
                break;

            case 'organization':
                skillModifiers.homeComfortMultiplier *= (1 + lvl * 0.025);
                skillModifiers.dailyExpenseMultiplier *= (1 - lvl * 0.015);
                break;

            case 'basicCreativity':
                skillModifiers.hobbyIncomeMultiplier *= (1 + lvl * 0.04);
                skillModifiers.positiveEventChanceBonus += lvl * 0.01;
                break;

            // ====================== ПРОФЕССИОНАЛЬНЫЕ НАВЫКИ ======================
            case 'professionalism':
                skillModifiers.salaryMultiplier *= (1 + lvl * 0.04);
                skillModifiers.workEfficiencyMultiplier *= (1 + lvl * 0.02);
                break;

            case 'leadership':
                skillModifiers.salaryMultiplier *= (1 + lvl * 0.02); // бонус за управление
                break;

            case 'negotiations':
                skillModifiers.salaryMultiplier *= (1 + lvl * 0.025);
                skillModifiers.shopPriceMultiplier *= (1 - lvl * 0.008);
                break;

            case 'stressResistance':
                skillModifiers.stressGainMultiplier *= (1 - lvl * 0.035);
                break;

            case 'cooking':
                skillModifiers.foodRecoveryMultiplier *= (1 + lvl * 0.07);
                skillModifiers.dailyExpenseMultiplier *= (1 - lvl * 0.03);
                break;

            case 'marketing':
                skillModifiers.hobbyIncomeMultiplier *= (1 + lvl * 0.09);
                skillModifiers.passiveIncomeBonus += lvl * 120;
                break;

            case 'specialization':
                skillModifiers.salaryMultiplier *= (1 + lvl * 0.07); // очень сильный бонус в своей сфере
                break;

            case 'technicalLiteracy':
                skillModifiers.learningSpeedMultiplier *= (1 + lvl * 0.06);
                skillModifiers.homeComfortMultiplier *= (1 + lvl * 0.04);
                break;

            case 'advancedHealth':
                skillModifiers.healthDecayMultiplier *= (1 - lvl * 0.025);
                skillModifiers.agingSpeedMultiplier *= (1 - lvl * 0.025);
                break;

            // ====================== ЛИЧНОСТНЫЕ И ТВОРЧЕСКИЕ ======================
            case 'charisma': // если добавишь харизму
                skillModifiers.relationshipGainMultiplier *= (1 + lvl * 0.06);
                break;

            case 'humor':
                skillModifiers.moodRecoveryMultiplier *= (1 + lvl * 0.05);
                break;

            // Добавляй новые навыки сюда по аналогии...

            default:
                // Для навыков, у которых пока нет специфической логики
                break;
        }
    });

    // Ограничиваем множители разумными пределами
    skillModifiers.salaryMultiplier = Math.max(0.5, Math.min(3.0, skillModifiers.salaryMultiplier));
    skillModifiers.shopPriceMultiplier = Math.max(0.4, Math.min(1.0, skillModifiers.shopPriceMultiplier));
    skillModifiers.stressGainMultiplier = Math.max(0.3, skillModifiers.stressGainMultiplier);

    console.log('✅ skillModifiers пересчитаны:', skillModifiers);
    return skillModifiers;
}

    playerSkills[skillKey] = newLevel;           // сохраняем уровень
    recalculateSkillModifiers(playerSkills);     // пересчитываем все модификаторы
}
